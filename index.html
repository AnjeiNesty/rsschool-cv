    function sortFructs(array) {
      let re = array.reduce((acc, cur) => {
        return {
            ...acc,
            [cur]: acc[cur] ? acc[cur] + 1 : 1
        }
      }, {});
      let sort = Object.keys(re).sort((a, b) => {
        return re[b] - re[a];
      });
      return sort;
    }
